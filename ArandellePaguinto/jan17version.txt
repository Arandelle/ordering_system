        // Render category tabs dynamically
        function renderCategoryTabs() {
            const tabsContainer = document.getElementById('categoryTabs');
            
            tabsContainer.innerHTML = categories.map((category, index) => {
                const itemCount = menuData[category.id] ? menuData[category.id].length : 0;
                const isActive = index === 0;
                
                return `
                    <button class="category-tab px-4 py-2 text-sm font-semibold whitespace-nowrap border-b-2 ${
                        isActive ? 'border-red-600 text-gray-900' : 'border-transparent text-gray-600 hover:text-gray-900'
                    }" data-category="${category.id}">
                        ${category.title} (${itemCount})
                    </button>
                `;
            }).join('');
        }

        // Render menu sections dynamically
        function renderMenuSections() {
            const sectionsContainer = document.getElementById('menuSections');
            
            sectionsContainer.innerHTML = categories.map((category, index) => {
                const items = menuData[category.id] || [];
                const isFirst = index === 0;
                
                return `
                    <div id="${category.id}Section" class="${isFirst ? 'scroll-mt-24' : 'mb-8 scroll-mt-24'}">
                        ${!isFirst ? `<h2 class="text-2xl font-bold mb-6">${category.title}</h2>` : ''}
                        <div id="${category.id}Items" class="grid grid-cols-1 md:grid-cols-2 gap-4 ${isFirst ? 'mb-12' : ''}">
                            ${items.map(item => createMenuItem(item)).join('')}
                        </div>
                    </div>
                `;
            }).join('');
        }

        // Render menu items
        function renderMenu() {
            renderCategoryTabs();
            renderMenuSections();
        }

        // Add to cart
        function addToCart(itemId) {
            // Get all items from all categories
            const allItems = Object.values(menuData).flat();
            const item = allItems.find(i => i.id === itemId);
            
            const existingItem = cart.find(i => i.id === itemId);
            if (existingItem) {
                existingItem.quantity++;
            } else {
                cart.push({ ...item, quantity: 1 });
            }
            
            updateCart();
        }

        // Remove from cart
        function removeFromCart(itemId) {
            cart = cart.filter(item => item.id !== itemId);
            updateCart();
        }

        // Update quantity
        function updateQuantity(itemId, change) {
            const item = cart.find(i => i.id === itemId);
            if (item) {
                item.quantity += change;
                if (item.quantity <= 0) {
                    removeFromCart(itemId);
                } else {
                    updateCart();
                }
            }
        }

        // Update cart UI
        function updateCart() {
            const cartItemsContainer = document.getElementById('cartItems');
            const cartTotal = document.getElementById('cartTotal');
            const emptyCart = document.getElementById('emptyCart');
            const summaryBtn = document.getElementById('summaryBtn');
            const checkoutBtn = document.getElementById('checkoutBtn');

            if (cart.length === 0) {
                cartItemsContainer.classList.add('hidden');
                emptyCart.classList.remove('hidden');
                summaryBtn.classList.add('hidden');
                cartTotal.textContent = '‚Ç± 0';
                checkoutBtn.classList.add('bg-gray-200', 'text-gray-400', 'cursor-not-allowed');
                checkoutBtn.classList.remove('bg-red-600', 'text-white', 'hover:bg-red-700', 'cursor-pointer');
            } else {
                cartItemsContainer.classList.remove('hidden');
                emptyCart.classList.add('hidden');
                summaryBtn.classList.remove('hidden');
                checkoutBtn.classList.remove('bg-gray-200', 'text-gray-400', 'cursor-not-allowed');
                checkoutBtn.classList.add('bg-red-600', 'text-white', 'hover:bg-red-700', 'cursor-pointer');

                const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
                cartTotal.textContent = `‚Ç± ${total}`;

                cartItemsContainer.innerHTML = cart.map(item => `
                    <div class="flex items-center justify-between py-3 border-b">
                        <div class="flex-1">
                            <p class="font-semibold text-sm">${item.name}</p>
                            <p class="text-xs text-gray-600">‚Ç± ${item.price}</p>
                        </div>
                        <div class="flex items-center gap-2">
                            <button 
                                onclick="updateQuantity(${item.id}, -1)"
                                class="w-6 h-6 bg-gray-200 rounded-full flex items-center justify-center hover:bg-gray-300">
                                -
                            </button>
                            <span class="w-6 text-center font-semibold">${item.quantity}</span>
                            <button 
                                onclick="updateQuantity(${item.id}, 1)"
                                class="w-6 h-6 bg-gray-200 rounded-full flex items-center justify-center hover:bg-gray-300">
                                +
                            </button>
                            <button 
                                onclick="removeFromCart(${item.id})"
                                class="ml-2 text-red-600 hover:text-red-800 text-sm">
                                √ó
                            </button>
                        </div>
                    </div>
                `).join('');
            }
        }

        // Category Navigation
        let categoryTabs;
        const categoryNav = document.getElementById('categoryNav');
        
        // Scroll to section when clicking category tab
        function initCategoryNavigation() {
            categoryTabs = document.querySelectorAll('.category-tab');
            
            categoryTabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    const category = this.getAttribute('data-category');
                    const section = document.getElementById(category + 'Section');
                    
                    if (section) {
                        const offset = 100; // Account for sticky nav
                        const elementPosition = section.getBoundingClientRect().top;
                        const offsetPosition = elementPosition + window.pageYOffset - offset;

                        window.scrollTo({
                            top: offsetPosition,
                            behavior: 'smooth'
                        });
                    }
                });
            });
        }

        // Update active category on scroll
        function updateActiveCategory() {
            if (!categoryTabs || categoryTabs.length === 0) return;
            
            const sections = categories.map(cat => ({
                id: cat.id,
                element: document.getElementById(cat.id + 'Section')
            })).filter(s => s.element);

            const scrollPosition = window.scrollY + 150;

            // Add shadow to nav when scrolled
            if (window.scrollY > 10) {
                categoryNav.classList.add('shadow-md');
            } else {
                categoryNav.classList.remove('shadow-md');
            }

            for (let i = sections.length - 1; i >= 0; i--) {
                const section = sections[i];
                if (section.element && section.element.offsetTop <= scrollPosition) {
                    // Remove active class from all tabs
                    categoryTabs.forEach(tab => {
                        tab.classList.remove('border-red-600', 'text-gray-900');
                        tab.classList.add('border-transparent', 'text-gray-600');
                    });

                    // Add active class to current tab
                    const activeTab = document.querySelector(`[data-category="${section.id}"]`);
                    if (activeTab) {
                        activeTab.classList.remove('border-transparent', 'text-gray-600');
                        activeTab.classList.add('border-red-600', 'text-gray-900');
                    }
                    break;
                }
            }
        }

        // Throttle scroll event for better performance
        let scrollTimeout;
        window.addEventListener('scroll', function() {
            if (scrollTimeout) {
                clearTimeout(scrollTimeout);
            }
            scrollTimeout = setTimeout(updateActiveCategory, 10);
        });

        // Initialize
        renderMenu();
        initCategoryNavigation();
    </script>
